Prevent screen blanking:

sudo nano /etc/kbd/config 
	BLANK_TIME=0
	POWERDOWN_TIME=0

sudo nano /etc/xdg/lxsession/LXDE-pi/autostart
	@xset s noblank 
	@xset s off 
	@xset -dpms


Add photobooth to auto start:
sudo crontab -e
	@reboot sh /home/pi/photobooth/launcher.sh >/home/pi/photobooth/cronlog 2>&1
